    pop ebx

    ; create file
    xor eax, eax
    xor ecx, ecx
    mov al, 0x8
    mov cx, 0q777
    int 0x80

    ;write data

    xor edx, edx
  
    push 0xA216465
    push 0x6b636148

    mov al, 0x4
    mov dl, 0x8
    int 0x80

    ; close file
    xor eax, eax
    mov al, 0x6
    int 0x80
    
    ; exit      
    xor ebx, ebx
    mov al, 0x1
    int 0x80                             
                                         
data:         
    call main_
